async function scrapeEvent(url) { eventBox.classList.add('visible'); eventBoxLabel.classList.add('loading'); eventBoxLabelText.textContent = 'Reading event page\u2026'; eventStatus.textContent = ''; eventDetailsEl.value = ''; eventDetailsEl.placeholder = 'Reading event details\u2026'; try { const res = await fetch(`/api/scrape-event?url=${encodeURIComponent(url)}`); const data = await res.json(); eventBoxLabel.classList.remove('loading'); if (data.success && data.details) { // Successfully extracted eventDetailsEl.value = data.details; eventBoxLabelText.textContent = 'Event details extracted \u2713 \u2014 edit if needed'; eventStatus.textContent = 'auto-filled'; } else if (data.template && data.details) { // Auth wall or JS-only page — pre-fill a template so user knows what to type eventDetailsEl.value = data.details; if (data.reason === 'auth_required') { eventBoxLabelText.textContent = '\ud83d\udd12 Login-protected page \u2014 fill in the details below'; eventStatus.textContent = 'manual entry'; } else { eventBoxLabelText.textContent = 'Page not readable \u2014 fill in the details below'; eventStatus.textContent = 'manual entry'; } // Focus and select so user can start typing immediately setTimeout(() => { eventDetailsEl.focus(); eventDetailsEl.select(); }, 100); } else { eventBoxLabelText.textContent = 'Could not read page \u2014 add details manually'; eventStatus.textContent = ''; eventDetailsEl.placeholder = 'e.g. Lego Robotics Summer Camp, ages 7\u201313, June 16\u201320, $249/week.'; } } catch { eventBoxLabel.classList.remove('loading'); eventBoxLabelText.textContent = 'Add event details manually'; eventDetailsEl.placeholder = 'e.g. Lego Robotics Summer Camp, ages 7\u201313, June 16\u201320, $249/week.'; } }
